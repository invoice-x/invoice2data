1"""Cross-page lines parsing (#650 / #624 / #359). 

2 

3Confirms the building blocks for multi-page invoices: 

4 

5- The lines parser already concatenates description values across multiple 

6 ``line`` matches (see ``parse_current_row``), so a multi-line description 

7 bracketed by ``first_line`` / ``last_line`` works as-is. 

8- ``skip_line`` (now pre-loop, in fd27734) filters repeating page headers / 

9 footers that would otherwise wrongly match ``first_line`` or ``line``. 

10- ``end_match: last`` makes ``parse_by_rule`` use the LAST occurrence of the 

11 ``end`` regex in the ``start``-bounded slice, so a block can span multiple 

12 pages where the ``end`` pattern repeats per-page (e.g. a footer). 

13""" 

14

42# === Description spanning multiple text lines (#624 / #359) === 

43 

44 

45def test_description_across_multiple_lines_concatenates() -> None: 

46 """`first_line` captures the SKU + start of description. 

47 

48 `line` captures description continuation; `last_line` captures the 

49 amounts. Description is concatenated with newlines (parse_current_row's 

50 default). 

51 """ 

52 settings = { 

53 "line_separator": r"\n", 

54 "first_line": r"(?m)^(?P<sku>SKU\d+)\s+(?P<name>.+)$", 

55 "line": r"(?m)^\s{4,}(?P<name>\w[\w\s]*)$", 

56 "last_line": ( 

57 r"(?m)^\s+(?P<qty>\d+)\s+(?P<price_unit>[\d.]+)\s+" 

58 r"(?P<price_subtotal>[\d.]+)$" 

59 ), 

60 } 

61 content = ( 

62 "SKU0001 Switch Pro Wireless Controller for\n" 

63 " Nintendo Switch Console Wireless\n" 

64 " Gamepad\n" 

65 " 40 18.50 648.00\n" 

66 "SKU0002 US/EU Wall Charger for DSi AC Adapter\n" 

67 " for Nintendo 3DS Power Supply for\n" 

68 " 3DSXL\n" 

69 " 1000 0.715 858.00\n" 

70 ) 

71 rows = lines_parser.parse_block(_TPL, "lines", settings, content) 

72 assert len(rows) == 2 

73 assert rows[0]["sku"] == "SKU0001" 

74 assert "Switch Pro Wireless Controller for" in rows[0]["name"] 

75 assert "Nintendo Switch Console Wireless" in rows[0]["name"] 

76 assert "Gamepad" in rows[0]["name"] 

77 assert rows[0]["qty"] == "40" 

78 assert rows[0]["price_subtotal"] == "648.00" 

79 assert rows[1]["sku"] == "SKU0002" 

80 assert "3DSXL" in rows[1]["name"]

83# === skip_line drops repeating page headers (#650) === 

84 

85 

86def test_skip_line_drops_repeating_page_headers() -> None: 

87 """A page-header line that would otherwise match `first_line` is dropped. 

88 

89 `skip_line` filters it out so the next real item is parsed correctly. 

90 """ 

91 settings = { 

92 "line_separator": r"\n", 

93 "line": r"^(?P<sku>SKU\d+)\s+(?P<name>.+)\s+(?P<qty>\d+)$", 

94 "skip_line": [ 

95 r"^Acme Corp \| Invoice continued$", 

96 r"^Page \d+ of \d+$", 

97 ], 

98 } 

99 content = ( 

100 "SKU0001 Widget 3\n" 

101 "Acme Corp | Invoice continued\n" 

102 "Page 1 of 2\n" 

103 "SKU0002 Gadget 7\n" 

104 "Page 2 of 2\n" 

105 "SKU0003 Gizmo 2\n" 

106 ) 

107 rows = lines_parser.parse_block(_TPL, "lines", settings, content) 

108 assert [r["sku"] for r in rows] == ["SKU0001", "SKU0002", "SKU0003"] 

109 assert [r["qty"] for r in rows] == ["3", "7", "2"]

112# === end_match: last bridges a per-page `end` pattern ===

114 

115def test_end_match_first_is_default_and_stops_at_first_end() -> None: 

116 """Default behavior: `end` matches the FIRST occurrence. 

117 

118 A per-page footer ends the block prematurely. 

119 """ 

120 settings = { 

121 "line_separator": r"\n", 

122 "start": r"(?m)^ITEMS$", 

123 "end": r"(?m)^_+\n\s*Total this page", 

124 "line": r"^(?P<sku>SKU\d+)\s+(?P<qty>\d+)$", 

125 } 

126 content = ( 

127 "ITEMS\n" 

128 "SKU0001 3\n" 

129 "SKU0002 5\n" 

130 "________________________________________\n" 

131 " Total this page: 8\n" 

132 "SKU0003 2\n" 

133 "________________________________________\n" 

134 " Total this page: 2\n" 

135 ) 

136 rows = lines_parser.parse_by_rule(_TPL, "lines", settings, content) 

137 # Only the first page parses; SKU0003 is past the (first) `end`. 

138 assert [r["sku"] for r in rows] == ["SKU0001", "SKU0002"]
