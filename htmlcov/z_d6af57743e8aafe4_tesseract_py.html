22def _imagemagick_cmd() -> list[str] | None: 

23 """Return the ImageMagick invocation prefix, or ``None`` when absent. 

24 

25 Prefers ``magick`` (ImageMagick 7+, cross-platform, and the only working 

26 variant on Windows where ``convert`` clashes with the built-in 

27 ``convert.exe``). Falls back to the legacy ``convert`` binary 

28 (ImageMagick 6) elsewhere. 

29 

30 Returns: 

31 list[str] | None: A prefix list ready to concatenate with the 

32 operation arguments, or ``None`` if ImageMagick isn't installed. 

33 """ 

34 if shutil.which("magick"): 

35 return ["magick"] 

36 if shutil.which("convert"): 

37 return ["convert"] 

38 return None
