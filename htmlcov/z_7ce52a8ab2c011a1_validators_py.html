1"""Lightweight, offline validators for disambiguating extracted identifiers. 

2 

3Field-type regexes overlap -- an IBAN and an EU VAT number can match each other's 

4pattern -- so a captured value is classified by *validating* it rather than by the 

5regex alone: try the strong checksum first (IBAN mod-97), then the format checks 

6(VAT, BIC). Pure-Python, no network, no heavy deps. 

7 

8Used by the template-authoring suggestion layer (to assign a captured value to the 

9right canonical field) and as an optional soft-validation of extracted fields. For 

10full per-country VAT/IBAN checksum coverage, ``python-stdnum`` can be layered on 

11later; this module stays dependency-free. 

12""" 

13 

14import re 

15 

16 

17# EU VAT number formats (full, including the country prefix). Greece uses "EL". 

18# Extend as needed -- this is a format check, not a per-country checksum. 

19_VAT_PATTERNS = { 

20 "AT": r"ATU\d{8}", 

21 "BE": r"BE[01]\d{9}", 

22 "BG": r"BG\d{9,10}", 

23 "CY": r"CY\d{8}[A-Z]", 

24 "CZ": r"CZ\d{8,10}", 

25 "DE": r"DE\d{9}", 

26 "DK": r"DK\d{8}", 

27 "EE": r"EE\d{9}", 

28 "EL": r"EL\d{9}", 

29 "ES": r"ES[A-Z0-9]\d{7}[A-Z0-9]", 

30 "FI": r"FI\d{8}", 

31 "FR": r"FR[A-Z0-9]{2}\d{9}", 

32 "HR": r"HR\d{11}", 

33 "HU": r"HU\d{8}", 

34 "IE": r"IE(?:\d{7}[A-Z]{1,2}|\d[A-Z]\d{5}[A-Z])", 

35 "IT": r"IT\d{11}", 

36 "LT": r"LT(?:\d{9}|\d{12})", 

37 "LU": r"LU\d{8}", 

38 "LV": r"LV\d{11}", 

39 "MT": r"MT\d{8}", 

40 "NL": r"NL\d{9}B\d{2}", 

41 "PL": r"PL\d{10}", 

42 "PT": r"PT\d{9}", 

43 "RO": r"RO\d{2,10}", 

44 "SE": r"SE\d{12}", 

45 "SI": r"SI\d{8}", 

46 "SK": r"SK\d{10}", 

47} 

48_VAT_RE = re.compile("|".join(f"(?:{pattern})" for pattern in _VAT_PATTERNS.values())) 

49_IBAN_RE = re.compile(r"[A-Z]{2}\d{2}[A-Z0-9]{11,30}") 

50_BIC_RE = re.compile(r"[A-Z]{6}[A-Z0-9]{2}(?:[A-Z0-9]{3})?") 

51 

52 

53def _normalize(value: str) -> str: 

54 """Strip whitespace and upper-case a candidate identifier. 

55 

56 Args: 

57 value (str): Raw captured value. 

58 

59 Returns: 

60 str: Whitespace-free, upper-cased value. 

61 """ 

62 return re.sub(r"\s+", "", value).upper() 

63 

64 

65def validate_iban(value: str) -> bool: 

66 """Return whether value is a structurally valid IBAN (ISO 13616 mod-97). 

67 

68 Args: 

69 value (str): Candidate IBAN; spaces and case are ignored. 

70 

71 Returns: 

72 bool: True if the structure and the mod-97 checksum are both valid. 

73 """ 

74 iban = _normalize(value) 

75 if not _IBAN_RE.fullmatch(iban): 

76 return False 

77 # Move the first four characters to the end, then map letters to numbers 

78 # (A=10 .. Z=35) and check the remainder mod 97 equals 1. 

79 rearranged = iban[4:] + iban[:4] 

80 digits = "".join(str(int(char, 36)) for char in rearranged) 

81 return int(digits) % 97 == 1 

82 

83 

84def validate_vat(value: str) -> bool: 

85 """Return whether value matches a known EU VAT number format. 

86 

87 Args: 

88 value (str): Candidate VAT number; spaces and case are ignored. 

89 

90 Returns: 

91 bool: True if it matches a supported country's VAT format. This is a 

92 format check, not a per-country checksum. 

93 """ 

94 return bool(_VAT_RE.fullmatch(_normalize(value))) 

95 

96 

97def validate_bic(value: str) -> bool: 

98 """Return whether value matches the SWIFT/BIC format (ISO 9362). 

99 

100 Args: 

101 value (str): Candidate BIC; spaces and case are ignored. 

102 

103 Returns: 

104 bool: True if it is a structurally valid 8- or 11-character BIC. 

105 """ 

106 return bool(_BIC_RE.fullmatch(_normalize(value))) 

107 

108 

109#: Validators in discriminating order -- strongest (checksum) first, so an IBAN is 

110#: never mistaken for a VAT number. 

111VALIDATORS = { 

112 "iban": validate_iban, 

113 "vat": validate_vat, 

114 "bic": validate_bic, 

115} 

116 

117 

118def classify_identifier(value: str) -> str | None: 

119 """Classify a captured value as one of the known identifier types. 

120 

121 Runs the validators in discriminating order (IBAN mod-97 checksum first, then 

122 the format-based VAT/BIC checks) so overlapping patterns resolve correctly -- 

123 e.g. a value that passes the IBAN checksum is reported as ``"iban"`` even 

124 though it may also look VAT-shaped. 

125 

126 Args: 

127 value (str): The captured string to classify. 

128 

129 Returns: 

130 str | None: The matching key ("iban", "vat" or "bic"), or None if nothing 

131 validates. 

132 """ 

133 for key, validator in VALIDATORS.items(): 

134 if validator(value): 

135 return key 

136 return None
