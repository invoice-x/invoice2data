47def parse_block( # noqa: RUF100 C901 

48 template: "InvoiceTemplate",

50 settings: dict[str, Any], 

51 content: str, 

52) -> list[dict[str, Any]]: 

53 """Parse a block of lines to extract data. 

54 

55 This function parses a block of lines from an invoice to extract data 

56 based on the provided template and settings. It handles different line 

57 types (first line, last line, regular lines) and can skip specific lines 

58 based on the configuration. 

59 

60 Args: 

61 template (InvoiceTemplate): The template containing extraction rules. 

62 field (str): The name of the field to extract. 

63 settings (dict[str, Any]): The settings for the extraction rule. 

64 content (str): The text content to parse. 

65 

66 Returns: 

67 list[dict[str, Any]]: A list of dictionaries, where each dictionary 

68 represents an extracted row with field-value pairs. 

69 """ 

70 # Validate settings 

71 assert "line" in settings, ( 

72 "Error in Template %s Line regex missing" % template["template_name"] 

73 ) 

74 

75 logger.debug("START lines block content ========================\n%s", content) 

76 logger.debug("END lines block content ==========================") 

77 lines: list[dict[str, Any]] = [] 

78 current_row: dict[str, Any] = {} 

79 

80 # We assume that structured line fields may either be individual lines or 

81 # they may be main line items with descriptions or details following beneath. 

82 # Using the first_line, line, last_line parameters we are able to capture this data. 

83 # first_line - The main line item 

84 # line - The detail lines, typically indented from the main lines 

85 # last_line - The final line of the indented lines 

86 # The code below will ignore both line and last_line until it finds first_line. 

87 # It will then switch to extracting line patterns until it either reaches last_line 

88 # or it reaches another first_line. 

89 

90 # As first_line and last_line are optional, if neither were provided, 

91 # set the first_line to be the provided line parameter. 

92 # In this way the code will simply loop through and extract the lines as expected. 

93 if "first_line" not in settings and "last_line" not in settings: 

94 settings["first_line"] = settings["line"] 

95 # As we enter the loop, we set the boolean for first_line being found to False, 

96 # This indicates the we are looking for the first_line pattern 

97 first_line_found = False 

98 skip_patterns = settings.get("skip_line") 

99 if skip_patterns and not isinstance(skip_patterns, list): 

100 skip_patterns = [skip_patterns] 

101 for line in _regex.split(settings["line_separator"], content): 

102 # If the line has empty lines in it , skip them 

103 if not line.strip("").strip("\n").strip("\r") or not line: 

104 continue 

105 # `skip_line: pattern` or `skip_line: [pat1, pat2]` lets a template drop 

106 # lines that match an unwanted shape (e.g. a sub-total / VAT footer that 

107 # the line regex would otherwise wrongly match). Issue #652. 

108 if skip_patterns and any( 

109 _regex.search(pattern, line) for pattern in skip_patterns 

110 ): 

111 logger.debug("skip_line match on %r", line) 

112 continue 

113 if "first_line" in settings: 113 ↛ 130line 113 didn't jump to line 130 because the condition on line 113 was always true

114 # Check if the current lines the first_line pattern 

115 match = parse_line(settings["first_line"], line) 

116 if match: 

117 # The line matches the first_line pattern so append current row to output 

118 # then assign a new current_row 

119 if current_row: 

120 lines.append(current_row) 

121 current_row = { 

122 field: value.strip() if value else "" 

123 for field, value in match.groupdict().items() 

124 } 

125 # Flip first_line_found boolean as first_line has been found 

126 # This will allow last_line and line to be matched on below 

127 first_line_found = True 

128 continue 

129 # Look for line or last_line only if the first has has been found 

130 if first_line_found is True: 

131 # If last_line was provided, check that 

132 if "last_line" in settings: 

133 # last_line pattern provided, so check if the current line is that line 

134 match = parse_line(settings["last_line"], line) 

135 if match: 

136 # This is the last_line, so parse all lines thus far, 

137 # append to output, 

138 # and reset current_row 

139 current_row = parse_current_row(match, current_row) 

140 if current_row: 140 ↛ 142line 140 didn't jump to line 142 because the condition on line 140 was always true

141 lines.append(current_row) 

142 current_row = {} 

143 # Flip first_line_found boolean to look for first_line again on next loop 

144 first_line_found = False 

145 continue 

146 # If none of those have continued the loop, check if this is just a normal line 

147 match = parse_line(settings["line"], line) 

148 if match: 

149 # This is one of the lines between first_line and last_line 

150 # Parse the data and add it to the current_row 

151 current_row = parse_current_row(match, current_row) 

152 continue 

153 # If the line doesn't match anything, log and continue to next line 

154 logger.debug("The following line doesn't match anything:\n*%s*", line) 

155 if current_row: 

156 # All lines processed, so append whatever the final current_row was to output 

157 lines.append(current_row)

169def _normalize_line_replace(spec: Any) -> dict[str, list[tuple[str, str]]]: 

170 """Normalize a lines ``replace`` setting to {sub-field: [(pattern, repl)]}. 

171 

172 Accepts a mapping ``{uom: [...], name: [...]}`` or a list of single-key 

173 mappings ``[{uom: [...]}, {name: [...]}]`` (each value being a single pair or 

174 a list of pairs, as for a field-level replace). 

175 

176 Args: 

177 spec (Any): The raw ``replace`` value from the lines settings. 

178 

179 Returns: 

180 dict[str, list[tuple[str, str]]]: Per-sub-field replacement pairs. 

181 """ 

182 if not spec: 

183 return {} 

184 raw: dict[str, Any] = {} 

185 if isinstance(spec, dict): 

186 raw = spec 

187 else: 

188 for entry in spec: 

189 raw.update(entry) 

190 return {field: _normalize_replacements(pairs) for field, pairs in raw.items()}

212def parse_by_rule( 

213 template: "InvoiceTemplate", 

214 field: str, 

215 rule: dict[str, Any], 

216 content: str, 

217) -> list[dict[str, Any]]: 

218 """Parse lines from a block of text based on a rule. 

219 

220 Args: 

221 template (InvoiceTemplate): The template dictionary. 

222 field (str): The field name. 

223 rule (dict[str, Any]): The rule dictionary. 

224 content (str): The text content to parse. 

225 

226 Returns: 

227 list[dict[str, Any]]: The parsed lines. 

228 """ 

229 # First apply default options. 

230 settings = DEFAULT_OPTIONS.copy() 

231 settings.update(rule) 

232 

233 # Validate settings 

234 assert "start" in settings, ( 

235 "Error in Template %s Lines start regex missing" % template["template_name"] 

236 ) 

237 assert "end" in settings, ( 

238 "Error in Template %s Lines end regex missing" % template["template_name"] 

239 ) 

240 

241 blocks_count = 0 

242 lines = [] 

243 end_match_strategy = settings.get("end_match", "first") 

244 if end_match_strategy not in ("first", "last"): 

245 logger.warning( 

246 "Template %s: end_match=%r is not 'first' or 'last'; defaulting to 'first'", 

247 template.get("template_name"), 

248 end_match_strategy, 

249 ) 

250 end_match_strategy = "first" 

251 

252 # Try finding & parsing blocks of lines one by one 

253 while True: 

254 start = _regex.search(settings["start"], content) 

255 if not start: 

256 logger.debug("Failed to find lines block start") 

257 break 

258 content = content[start.end() :] 

259 

260 if end_match_strategy == "last": 

261 # Cross-page recipe: if `end` matches a per-page footer (e.g. a 

262 # repeated total/separator block), use the LAST match in this 

263 # `start`-bounded slice so the block can span all pages. 

264 end_matches = list(_regex.finditer(settings["end"], content)) 

265 end = end_matches[-1] if end_matches else None 

266 else: 

267 end = _regex.search(settings["end"], content) 

268 if not end: 

269 logger.debug("Failed to find lines block end") 

270 break 

271 

272 blocks_count += 1 

273 lines += parse_block(template, field, settings, content[0 : end.start()]) 

274 

275 content = content[end.end() :] 

276 

277 if blocks_count == 0: 

278 logger.warning( 

279 'Failed to find any matching block (part) of invoice for "%s"', field 

280 ) 

281 elif not lines: 

282 logger.warning('Failed to find any lines for "%s"', field) 

283 

284 return lines
