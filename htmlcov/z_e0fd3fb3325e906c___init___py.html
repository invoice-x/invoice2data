11#: ``--output-name`` values that stream to a standard stream instead of a file. 

12#: ``/dev/stdout`` / ``/dev/stderr`` are treated as portable aliases for the 

13#: Python streams (so they also work on platforms without those device files). 

14_STDOUT_ALIASES = frozenset({"-", "/dev/stdout"}) 

15_STDERR_ALIASES = frozenset({"/dev/stderr"}) 

16 

17 

18@contextlib.contextmanager 

19def open_output(path: str, suffix: str, **open_kwargs: Any) -> Iterator[TextIO]: 

20 """Open the destination for a given ``--output-name``. 

21 

22 Streams to stdout for ``-`` / ``/dev/stdout`` and stderr for ``/dev/stderr``; 

23 otherwise writes to ``path`` with ``suffix`` appended when missing. 

24 

25 Args: 

26 path (str): The requested output name. 

27 suffix (str): File extension to ensure (e.g. ``.json``) for file paths. 

28 **open_kwargs (Any): Extra arguments forwarded to ``Path.open`` for files. 

29 

30 Yields: 

31 TextIO: The writable text stream (a standard stream is not closed here). 

32 """ 

33 if path in _STDOUT_ALIASES: 

34 yield sys.stdout 

35 elif path in _STDERR_ALIASES: 

36 yield sys.stderr 

37 else: 

38 filename = path if path.endswith(suffix) else path + suffix 

39 with Path(filename).open("w", **open_kwargs) as handle: 

40 yield handle
